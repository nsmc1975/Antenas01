159Antena* removerAntena(Antena* inicio, int x, int y) {
160 if (inicio == NULL) {
161 printf("Erro: Lista vazia, não há antenas para remover.\n");
162 return NULL;
163 }
164
165 Antena* atual = inicio;
166 Antena* anterior = NULL;
167
168 // Percorrer a lista para encontrar a antena a remover
169 while (atual != NULL) {
170 if (atual->x == x && atual->y == y) {
171 // Encontrou a antena a remover
172 if (anterior == NULL) {
173 // Caso especial: a antena está no início da lista
174 inicio = atual->prox;
175 }
176 else {
177 // Caso geral: ligar o anterior ao próximo nó
178 anterior->prox = atual->prox;
179 }
180 free(atual); // Libertar a memória da antena removida
181 printf("Antena em (%d, %d) removida com sucesso.\n", x, y);
182 return inicio;
183 }
184 anterior = atual;
185 atual = atual->prox;
186 }
187
188 // Se chegou aqui, significa que não encontrou a antena
189 printf("Erro: Nenhuma antena encontrada em (%d, %d).\n", x, y);
190 return inicio;
191}
